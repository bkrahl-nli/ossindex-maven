46  /**
47   * Enforcer rule to ban vulnerable dependencies.
48   *
49   * @since ???
50   */
51  public class BanVulnerableDependencies
52      extends EnforcerRuleSupport
53  {
54    private OssindexClientConfiguration clientConfiguration = new OssindexClientConfiguration();
55  
56    private String scope;
57  
58    private boolean transitive = true;
59  
60    private Set<MavenCoordinates> excludeCoordinates = new HashSet<>();
61  
62    private float cvssScoreThreshold = 0;
63  
64    private Set<String> excludeVulnerabilityIds = new HashSet<>();
65  
66    /**
67     * <a href="https://ossindex.sonatype.org/">Sonatype OSS Index</a> client configuration.
68     */
69    @SuppressWarnings("unused")
70    public void setClientConfiguration(final OssindexClientConfiguration clientConfiguration) {
71      this.clientConfiguration = clientConfiguration;
72    }
73  
74    /**
75     * Limit scope of dependency resolution.
76     */
77    @SuppressWarnings("unused")
78    public void setScope(final String scope) {
79      this.scope = scope;
80    }
81  
82    /**
83     * Include transitive dependencies.
84     */
85    @SuppressWarnings("unused")
86    public void setTransitive(final boolean transitive) {
87      this.transitive = transitive;
88    }
89  
90    /**
91     * Set of coordinates to exclude from vulnerability matching.
92     */
93    @SuppressWarnings("unused")
94    public void setExcludeCoordinates(final Set<MavenCoordinates> excludeCoordinates) {
95      this.excludeCoordinates = excludeCoordinates;
96    }
97  
98    /**
99     * CVSS-score threshold.
100    *
101    * Vulnerabilities with lower scores will be excluded.
102    */
103   @SuppressWarnings("unused")
104   public void setCvssScoreThreshold(final float cvssScoreThreshold) {
105     this.cvssScoreThreshold = cvssScoreThreshold;
106   }
107 
108   /**
109    * Set of <a href="https://ossindex.sonatype.org/">Sonatype OSS Index</a>
110    * vulnerability identifiers to exclude from matching.
111    */
112   @SuppressWarnings("unused")
113   public void setExcludeVulnerabilityIds(final Set<String> excludeVulnerabilityIds) {
114     this.excludeVulnerabilityIds = excludeVulnerabilityIds;
115   }
116 
117   @Override
118   public void execute(@Nonnull final EnforcerRuleHelper helper) throws EnforcerRuleException {
119     new Task(helper).run();
120   }
121 
122   /**
123    * Encapsulates state for rule evaluation.
124    */
125   private class Task
126   {
127     private final Log log;
128 
129     private final MavenSession session;
130 
131     private final MavenProject project;
132 
133     private final Settings settings;
134 
135     private final DependencyGraphBuilder graphBuilder;
136 
137     private final ComponentReportAssistant reportAssistant;
138 
139     public Task(final EnforcerRuleHelper helper) {
140       this.log = helper.getLog();
141       this.session = lookup(helper, "${session}", MavenSession.class);
142       this.project = lookup(helper, "${project}", MavenProject.class);
143       this.settings = lookup(helper, "${settings}", Settings.class);
144       this.graphBuilder = lookup(helper, DependencyGraphBuilder.class);
145       this.reportAssistant = lookup(helper, ComponentReportAssistant.class);
146     }
147 
148     public void run() throws EnforcerRuleException {
149       // skip if maven is in offline mode
150       if (session.isOffline()) {
151         log.warn(skipReason("offline"));
152         return;
153       }
154 
155       // skip if packaging is pom
156       if ("pom".equals(project.getPackaging())) {
157         log.debug(skipReason("POM module"));
158         return;
159       }
160 
161       // determine dependencies
162       Set<Artifact> dependencies;
163       try {
164         dependencies = resolveDependencies();
165       }
166       catch (DependencyGraphBuilderException e) {
167         throw new RuntimeException("Failed to resolve dependencies", e);
168       }
169 
170       // skip if project has no dependencies
171       if (dependencies.isEmpty()) {
172         log.debug(skipReason("zero dependencies"));
173         return;
174       }
175 
176       // adapt maven http-proxy settings to client configuration
177       maybeApplyProxy(clientConfiguration);
178 
179       ComponentReportRequest reportRequest = new ComponentReportRequest();
180       reportRequest.setClientConfiguration(clientConfiguration);
181       reportRequest.setExcludeCoordinates(excludeCoordinates);
182       reportRequest.setExcludeVulnerabilityIds(excludeVulnerabilityIds);
183       reportRequest.setCvssScoreThreshold(cvssScoreThreshold);
184       reportRequest.setComponents(dependencies);
185 
186       ComponentReportResult reportResult = reportAssistant.request(reportRequest);
187 
188       if (reportResult.hasVulnerable()) {
189         throw new EnforcerRuleException(reportResult.explain());
190       }
191     }
192 
193     /**
194      * Generate skip reason message; includes rule simple-name for clarity.
195      */
196     private String skipReason(final String reason) {
197       return String.format("Skipping %s; %s", BanVulnerableDependencies.class.getSimpleName(), reason);
198     }
199 
200     //
201     // HTTP-proxy support; some duplication due to maven-version mismatch
202     //
203 
204     /**
205      * Detect proxy configuration from Maven settings.
206      */
207     @SuppressWarnings("Duplicates")
208     @Nullable
209     private ProxyConfiguration detectProxy() {
210       Proxy proxy = settings.getActiveProxy();
211       if (proxy != null) {
212         ProxyConfiguration config = new ProxyConfiguration();
213         config.setProtocol(proxy.getProtocol());
214         config.setHost(proxy.getHost());
215         config.setPort(proxy.getPort());
216         config.setNonProxyHosts(proxy.getNonProxyHosts());
217 
218         // maybe add authentication if username & password are present
219         String username = Strings.emptyToNull(proxy.getUsername());
220         String password = Strings.emptyToNull(proxy.getPassword());
221         if (username != null && password != null) {
222           config.setAuthConfiguration(new AuthConfiguration(username, password));
223         }
224         return config;
225       }
226       return null;
227     }
228 
229     /**
230      * If a proxy configuration was detected then configure client, unless client has this configured already.
231      */
232     private void maybeApplyProxy(final OssindexClientConfiguration clientConfiguration) {
233       if (clientConfiguration.getProxyConfiguration() == null) {
234         ProxyConfiguration proxy = detectProxy();
235         if (proxy != null) {
236           clientConfiguration.setProxyConfiguration(proxy);
237         }
238       }
239     }
240 
241     //
242     // Dependency resolution; some duplication due to maven-version mismatch
243     //
244 
245     /**
246      * Resolve dependencies to inspect for vulnerabilities.
247      */
248     private Set<Artifact> resolveDependencies() throws DependencyGraphBuilderException {
249       Set<Artifact> result = new HashSet<>();
250 
251       ArtifactFilter artifactFilter = null;
252       if (scope != null) {
253         List<String> scopes = Splitter.on(',').trimResults().omitEmptyStrings().splitToList(scope);
254         artifactFilter = new CumulativeScopeArtifactFilter(scopes);
255       }
256 
257       DependencyNode node = graphBuilder.buildDependencyGraph(project, artifactFilter);
258       collectArtifacts(result, node);
259 
260       return result;
261     }
262 
263     /**
264      * Collect artifacts from dependency.
265      *
266      * Optionally including transitive dependencies if {@link #transitive} is {@code true}.
267      */
268     @SuppressWarnings("Duplicates")
269     private void collectArtifacts(final Set<Artifact> artifacts, final DependencyNode node) {
270       if (node.getChildren() != null) {
271         for (DependencyNode child : node.getChildren()) {
272           artifacts.add(child.getArtifact());
273 
274           if (transitive) {
275             collectArtifacts(artifacts, child);
276           }
277         }
278       }
279     }
280   }
